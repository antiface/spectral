[docs]class Element(object):

    def __init__(self, key, title):
        self.uuid = uuid.uuid4().hex
        self.key = key
        self.title = title
        self.width = 1

        self.context = {}
        self.template = ""

    @property
    def js_init(self):
        return ""




[docs]class TextElement(Element):

    def __init__(self, key, title=None, value=None):
        super(TextElement, self).__init__(key, title)
        self.value = value

        self.context = {
            'title': self.title,
            'id': self.uuid
        }
        self.template = "element_text.html"

    @property
    def js_init(self):
        return "Control.register(new TextElement('{0}', '{1}'));".format(self.uuid, self.key)
